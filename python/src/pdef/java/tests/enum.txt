package module;

import pdef.EnumDescriptor;
import pdef.EnumType;
import pdef.generated.GeneratedEnumDescriptor;

public enum ObjectType implements EnumType {
    OBJECT,USER;

	@Override
	public EnumDescriptor getDescriptor() {
		return descriptor;
	}

	private static final EnumDescriptor descriptor = new GeneratedEnumDescriptor(ObjectType.class);
	public static EnumDescriptor getDescriptorForType() {
		return descriptor;
	}
}
